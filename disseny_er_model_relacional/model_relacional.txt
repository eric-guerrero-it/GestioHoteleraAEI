
-- MODEL RELACIONAL: Cadena hotelera ESPAMUS+. Grup12 AEI

-- Informació bàsica dels hotels gestionats
HOTEL(idHotel, nom, estrelles, adreça, població, web, telèfon)

-- Dades comunes de totes les persones (clients i treballadors)
PERSONA(DNI, nom, cognoms, telèfon, adreça, dataNaixement)

-- Clients derivats de PERSONA
CLIENT(DNI)
ON {DNI} REFERENCIA PERSONA {DNI}

-- Treballadors derivats de PERSONA
TREBALLADOR(DNI, tipusTreballador)
ON {DNI} REFERENCIA PERSONA {DNI}

-- Subtipus de TREBALLADOR: recepció
RECEPCIO(DNI, anysExperiència)
ON {DNI} REFERENCIA TREBALLADOR {DNI}

-- Subtipus de TREBALLADOR: cuina, amb relació reflexiva revisatPer
CUINA(DNI, categoria, darrerLlocTreball, revisatPer)
ON {DNI} REFERENCIA TREBALLADOR {DNI}
ON {revisatPer} REFERENCIA CUINA {DNI}

-- Subtipus genèric per altres treballadors
RESTAPersonal(DNI)
ON {DNI} REFERENCIA TREBALLADOR {DNI}

-- Entitat d’idiomes reconeguts
IDIOMA(nom)

-- Coneixements d’idiomes per treballador
CONEIXEMENT(DNI, nomIdioma, parla, entén, escriu)
ON {DNI} REFERENCIA TREBALLADOR {DNI}
ON {nomIdioma} REFERENCIA IDIOMA {nom}
PK: {DNI, nomIdioma}

-- Serveis oferts pels hotels
SERVEI(idServei, nom, cost)

-- Relació N:M entre hotels i serveis
HOTEL_SERVEI(idHotel, idServei)
ON {idHotel} REFERENCIA HOTEL {idHotel}
ON {idServei} REFERENCIA SERVEI {idServei}
PK: {idHotel, idServei}

-- Habitacions dels hotels
HABITACIO(idHabitació, número, llits, m2, teNevera, teTelevisió, idHotel)
ON {idHotel} REFERENCIA HOTEL {idHotel}

-- Activitats ofertes pels hotels
ACTIVITAT(idActivitat, idHotel, nom, descripció, data, preu)
ON {idHotel} REFERENCIA HOTEL {idHotel}

-- Assignació de treballadors a un hotel
TREBALLA(idHotel, DNI_treballador)
ON {idHotel} REFERENCIA HOTEL {idHotel}
ON {DNI_treballador} REFERENCIA TREBALLADOR {DNI}

-- Reserves fetes per clients
RESERVA(idReserva, dataInici, dataFinal, idHotel, DNI_client)
ON {idHotel} REFERENCIA HOTEL {idHotel}
ON {DNI_client} REFERENCIA CLIENT {DNI}

-- Habitacions assignades a reserves
RESERVA_HABITACIO(idReserva, idHabitació, preuTemporadaAlta, preuTemporadaBaixa)
ON {idReserva} REFERENCIA RESERVA {idReserva}
ON {idHabitació} REFERENCIA HABITACIO {idHabitació}
PK: {idReserva, idHabitació}

-- Sol·licituds de serveis durant l'estada
SOLLICITUD(idSollicitud, dataHora, idHotel, idServei, DNI_client, pagarEnCheckOut)
ON {idHotel} REFERENCIA HOTEL {idHotel}
ON {idServei} REFERENCIA SERVEI {idServei}
ON {DNI_client} REFERENCIA CLIENT {DNI}

-- Facturació generada a partir de les reserves i serveis
FACTURA(idFactura, dataEmissió, importTotal, DNI_client)
ON {DNI_client} REFERENCIA CLIENT {DNI}

-- Relació N:M entre factures i serveis consumits
FACTURA_SERVEI(idFactura, idSollicitud)
ON {idFactura} REFERENCIA FACTURA {idFactura}
ON {idSollicitud} REFERENCIA SOLLICITUD {idSollicitud}
PK: {idFactura, idSollicitud}

-- Registre dels enviaments a l'API externa (p. ex., Mossos d'Esquadra)
ENVIAMENT_API(id, data_hora, estat, json_enviat)
